# nlogn算法

Merge Sort归并排序法

将数组拆分成单个后进行两两排序，接着四四排序，接着八八排序，逐步合并到整体合并

const left = [1,3,5,7];
const right = [2,4,6,8];
let arr = new Array(8);
let cur = 0;
let i = 0;
let j = 0;
for(let cur = 0; cur < arr.length ; cur ++){
    if(left[i]<right[j]){
        arr[cur] = left[i];
        i++;
    }else{
        arr[cur] = right[j];
        j++;
    }
};
console.log(arr);

# 随机数组排序

let arr,newArr,i,temp;
arr = [1,8,3,6,8,0,9,8,7,2,9,5,4,6,5];
newArr = [];
arr.forEach(o=>{
	if(!newArr.length) return newArr.push(o);
	temp = '';
	i = 0;
	newArr.forEach(k=>{
		i++;
		if(o>k) {
			temp = i-1;
        }
		if(i == newArr.length){
			if(temp === '') {
				newArr.unshift(o);
			}else {
				newArr.splice(temp+1,0,o);
			}
		}
	})
});
